    \exercise{4.1: SQL und Relationenalgebra}

    \subsection*{a) Relationale Algebra: Kunden jünger als 40 Jahre}
% Einfache Selektion auf das Alter, gefolgt von einer Projektion auf den Namen.
    \[
        \pi_{k\_name}(\sigma_{k\_alter < 40}(kunde))
    \]

    \subsection*{b) Relationale Algebra: Bankkonten mit hohem Darlehen und Kontaktperson nicht in Berlin}
% Um die Attribute aus darlehen, bankkonto, mitarbeiter und adresse zu verknüpfen,
% nutzen wir Theta-Joins. Unnötige Tabellen (wie kunde) werden weggelassen.
% Vor den Joins filtern wir die Tabellen darlehen und adresse direkt, um die Datenmenge
% frühzeitig zu reduzieren (Push-Down-Prinzip).
    \[
        \pi_{b\_id} \left(
                        \sigma_{d\_kredithoehe > 30000}(darlehen)
                        \bowtie_{d\_bankkonto = b\_id}
                        bankkonto
                        \bowtie_{b\_kontaktperson = m\_id}
                        mitarbeiter
                        \bowtie_{m\_adresse = a\_id}
                        \sigma_{a\_ort \neq \text{'Berlin'}}(adresse)
        \right)
    \]

    \subsection*{c) SQL: Name, Alter, PLZ aller Kunden, aufsteigend sortiert nach Alter}
% Einfacher Join zwischen Kunde und Adresse in der WHERE-Klausel.
    \begin{lstlisting}[language=SQL]
SELECT kunde.k_name, kunde.k_alter, adresse.a_plz
FROM kunde, adresse
WHERE kunde.k_adresse = adresse.a_id
ORDER BY kunde.k_alter ASC;
    \end{lstlisting}

    \subsection*{d) SQL: Mitarbeiter mit mehr als 10 qualifizierten Bankkonten}
% Hier müssen wir die Mitarbeiter mit den Bankkonten verknüpfen und direkt auf
% das Guthaben (> 20000) filtern. Anschließend wird über die Mitarbeiter-ID gruppiert.
% WICHTIG: Alles, was im SELECT steht (außer den Aggregationen wie COUNT),
% MUSS im GROUP BY stehen (daher m_id und m_name).
    \begin{lstlisting}[language=SQL]
SELECT mitarbeiter.m_name, COUNT(bankkonto.b_id)
FROM mitarbeiter, bankkonto
WHERE mitarbeiter.m_id = bankkonto.b_kontaktperson
  AND bankkonto.b_guthaben > 20000
GROUP BY mitarbeiter.m_id, mitarbeiter.m_name
HAVING COUNT(bankkonto.b_id) > 10;
    \end{lstlisting}

    \subsection*{e) SQL: Kunden mit Darlehen über dem Durchschnitt}
% Da die Tabelle "darlehen" glücklicherweise einen direkten Fremdschlüssel
% "d_kunde" besitzt, ist ein überflüssiger Join über "bankkonto" nicht nötig!
% DISTINCT verhindert doppelte Einträge, falls ein Kunde mehrere solche Darlehen hat.
    \begin{lstlisting}[language=SQL]
SELECT DISTINCT kunde.k_name, kunde.k_alter
FROM kunde, darlehen
WHERE kunde.k_id = darlehen.d_kunde
  AND darlehen.d_kredithoehe > (
      SELECT AVG(darlehen.d_kredithoehe)
      FROM darlehen
  );
    \end{lstlisting}

    \subsection*{f) SQL: Adressen-Ortsnamen standardisieren ('Koeln...')}
% UPDATE-Befehl mit dem LIKE-Operator. Das Prozentzeichen (%) fungiert als Wildcard
% für beliebig viele, beliebige Zeichenfolgen nach dem Wort 'Koeln'.
    \begin{lstlisting}[language=SQL]
UPDATE adresse
SET a_ort = 'Koeln'
WHERE a_ort LIKE 'Koeln%';
    \end{lstlisting}
